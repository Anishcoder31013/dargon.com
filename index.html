#include <stdio.h>
int fun1(int array[],int n)
{
int sum=0;

for(int i=0;i<n;i++)
{
sum = sum + array[i];

}
return sum;
}
float fun2(int array[],int n)
{
int a = fun1(array,n);
float avg = (float)a/(float)n;
return avg;
}
float fun3(int array[],int n,int max)
{
int a = fun1(array,n);
int b = max*n;
float avgMarks = (float)a/(float)b*100;
return avgMarks;
}
int main(int argc, char *argv[])
{
int n;
int max;
printf("ENTER TOTAL NO OF STUDENTS:\n");
scanf("%d",&n);
int array[n];
printf("ENTER MAXIMUM MARKS:\n");
scanf("%d",&max);
for(int i=0;i<n;i++)
{
printf("ENTER MARKS OF STUDENT %d:\n",i+1);
scanf("%d",&array[i]);
}
int c= fun1(array,n);
printf("THE SUM OF MARKS OF ALL THE STUDENTS IS %d\n",c)
float d=fun2(array,n);
printf("THE AVERAGE OF MARKS OF ALL THE STUDENTS IS %.2f\n",d);
float e = fun3(array,n,max);
printf("AVERAGE PERCENTAGE OF ALL THE STDENTS IS %.2f%\n",e);
}
